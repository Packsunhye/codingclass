.loader  {
	position: absolute;
	top: 50%;
	left:50%; 
	transform:translate(-50%,-50%); 
	width:100px; 
	height:100px; 
	animation: spin .6s linear infinite reverse;
}

	content: ''; 
	background-color: #fff;
	width:5px; 
	height:5px;
	border-radius:100%; 
	top:0px;
}
