02. 나의 문제 풀이 

function solution(n) {
    for(let x = 2; x < n; x++){
        if( n % x === 1 ) return x;
    }   
}

03. 다른 사람의 문제 풀이 

function solution(n) {
    let answer = 0;
    for (let i = 0; i < n; i++) {
        if (answer > 0) {
            break;
        }

        if (n % i == 1) {
            answer = i;    
        }
    }

    return answer;
}
