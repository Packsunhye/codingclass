02. 나의 문제 풀이 

function solution(num) {
    var answer = '';
    if( num % 2 == 0){
        return answer = "Even";
    }else {
        return answer = "Odd";
    }
}

03. 다른 사람의 문제 풀이 

function evenOrOdd(num) {
    return num % 2 ? "Odd" : "Even";
}
