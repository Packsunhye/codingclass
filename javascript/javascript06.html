번호 기본값 메서드 리턴값

        document.querySelector(".sample01_M1").innerHTML = "length";
        document.querySelector(".sample01_P1").innerHTML = arrNum.length;

        //02

        //03

        document.querySelector(".sample01_N3").innerHTML = "3";

03. 배열 메서드 : push() | pop()

push()는 배열 마지막 요소에 숫자를 추가하고 pop()은 배열 마지막 요소를 삭제된 요소를 반환한다.

번호 기본값 메서드 리턴값 결과값 
        //01  
        const arrNum = [100,200,300,400,500];
        const arrPush = arrNum.push(600);
    
        document.querySelector(".sample03_N1").innerHTML = "1";
        document.querySelector(".sample03_Q1").innerHTML = "[100,200,300,400,500]";
        document.querySelector(".sample03_M1").innerHTML = "push(600)";
        document.querySelector(".sample03_P1").innerHTML = arrPush; //600을 추가하고 숫자를 셈
        document.querySelector(".sample03_A1").innerHTML = arrNum; //600을 추가하고 한 글자가 됨
    
        //02
        const arrNum2 = [100,200,300,400,500];
        const arrPush2 = arrNum2.pop();
    
        document.querySelector(".sample03_N2").innerHTML = "2";
        document.querySelector(".sample03_Q2").innerHTML = "[100,200,300,400,500]";
        document.querySelector(".sample03_M2").innerHTML = "pop()";
        document.querySelector(".sample03_P2").innerHTML = arrPush2; //500을 삭제하기 때문에 500이 나와야함
        document.querySelector(".sample03_A2").innerHTML = arrNum2; //500이 삭제 되었기 때문에 100,200,300,400이 나옴

04. 배열 메서드 : unshift() / shift()

unshift()는 배열 처음 요소에 추가하고 shift()은 배열 처음 요소에 삭제한다.

번호 기본값 메서드 리턴값 결과값 
        const arrNum = [100,200,300,400,500];
        const arrUnshift = arrNum.unshift(600);
        
        document.querySelector(".sample04_N1").innerHTML = "1";
        document.querySelector(".sample04_Q1").innerHTML = "[100,200,300,400,500]";
        document.querySelector(".sample04_M1").innerHTML = "unshift(600)";
        document.querySelector(".sample04_P1").innerHTML = arrUnshift; // 600을 맨 앞에 추가해서, 숫자 6이 나온다
        document.querySelector(".sample04_A1").innerHTML = arrNum; //600,100,200,300,400,500이 나온다.
        
        const arrNum2 = [100,200,300,400,500];
        const arrShift = arrNum2.shift();
        
        document.querySelector(".sample04_N2").innerHTML = "2";
        document.querySelector(".sample04_Q2").innerHTML = "[100,200,300,400,500]";
        document.querySelector(".sample04_M2").innerHTML = "shift()";
        document.querySelector(".sample04_P2").innerHTML = arrShift; //맨 앞의 숫자를 삭제하기 때문에 100이 나온다.
        document.querySelector(".sample04_A2").innerHTML = arrNum2; //100을 삭제해서 200,300,400,500 이 나온다.
