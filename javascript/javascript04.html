03. 다중 if문

   const num = 100;
    
    if ( num == 90 ){
        document.write("실행되었습니다.(num == 90)");
    } else if (num == 100) {
        document.write("실행되었습니다.(num == 100)");
    } else if (num == 110) {
        document.write("실행되었습니다.(num == 110)");
    } else if (num == 120) {
        document.write("실행되었습니다.(num == 120)");
    } else {
        document.write("실행되었습니다.(num == 값이 없음)");
    }
결과보기
실행되었습니다.(num == 100)

06. swich문

   const num = 100;
    
    switch(num){
        case 100 : 
            document.write("실행되었습니다.num==100)");
            break;
        case 110 : 
            document.write("실행되었습니다.num==110)");
            break;   
        case 120 : 
            document.write("실행되었습니다.num==120)");
            break;   
        case 130 : 
            document.write("실행되었습니다.num==130)");
            break; 
        default :
                document.write("실행되었습니다.num==값이 없음)"); 
    }
결과보기
실행되었습니다.(num==100)
